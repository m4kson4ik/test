1.Изменить все даты в таблице Заказы на 28 лет
UPDATE Заказы
SET ДатаРазмещения = DATEADD(year, 28, ДатаРазмещения),
    ДатаНазначения = DATEADD(year, 28, ДатаНазначения),
    ДатаИсполнения = DATEADD(year, 28, ДатаИсполнения)

2. Вывести количество товаров на складе, поставленных каждым поставщиком
SELECT p.название AS Поставщик, SUM(т.наСкладе) AS Количество_товаров
FROM dbo.Товары т
INNER JOIN dbo.Поставщики p ON т.кодПоставщика = p.кодПоставщика
GROUP BY p.название

3. Вывести количество товаров в каждой категории
SELECT т.кодТипа, COUNT(*) AS Количество_товаров
FROM dbo.Товары т
INNER JOIN dbo.Типы тип ON т.кодТипа = тип.кодТипа
GROUP BY т.кодТипа

4. Выбрать категории товаров, которые заказывают чаще всего
SELECT Типы.описание, COUNT(*) AS Количество_заказов
FROM dbo.Заказы
INNER JOIN dbo.Заказано ON Заказы.КодЗаказа = Заказано.КодЗаказа
INNER JOIN dbo.Товары ON Заказано.КодТовара = Товары.КодТовара
INNER JOIN dbo.Типы ON Товары.кодТипа = Типы.кодТипа
GROUP BY Типы.описание
ORDER BY COUNT(*) DESC

5. Вывести количество проданных товаров в каждой категории по годам
SELECT YEAR(Заказы.ДатаИсполнения) AS Год, Типы.описание AS Категория, SUM(Заказано.количество) AS Продано
FROM dbo.Заказы
INNER JOIN dbo.Заказано ON Заказы.КодЗаказа = Заказано.КодЗаказа
INNER JOIN dbo.Товары ON Заказано.КодТовара = Товары.КодТовара
INNER JOIN dbo.Типы ON Товары.кодТипа = Типы.кодТипа
GROUP BY YEAR(Заказы.ДатаИсполнения), Типы.описание

6. Вывести название страны, поставившей больше всего продуктов по количеству
SELECT TOP 1 p.Страна, COUNT(z.КодТовара) AS КоличествоПродуктов
FROM dbo.Поставщики p
JOIN dbo.Товары t ON p.кодПоставщика = t.кодПоставщика
JOIN dbo.Заказано z ON t.кодТовара = z.кодТовара
GROUP BY p.Страна
ORDER BY КоличествоПродуктов DESC;

7. Вывести список товаров с ценой, выше средней
SELECT *
FROM dbo.Товары
WHERE цена > (SELECT AVG(цена) FROM dbo.Товары)
